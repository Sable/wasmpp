13 // RelocMat is useful for a matrix which can have a hybrid starting address.
14 // A matrix wrapped in this class should be used carefully because
15 // a begin address should point to a matrix with the exact shape
16 // as the passed array, otherwise the program will result in an undefined
17 // behaviour
18 class RelocMat {
19 public:
20  RelocMat(ds::NDArray* array) : array_(array), has_begin_var(false) {}
21  RelocMat(ds::NDArray* array, wabt::Var var) : array_(array), var_(var), has_begin_var(true) {}
22  ds::NDArray* Array() const { return array_; }
23  wabt::Var Var() const { assert(has_begin_var); return var_; }
24  bool HasBeginVar() const { return has_begin_var; }
25 private:
26  ds::NDArray* array_;
27  wabt::Var var_;
28  bool has_begin_var = false;
29 };
